# extract euclidean distance between start and end locations
# Convert latitude and longitude from degrees to radians
def calculate_euclidean_distances(lat1, lng1, lat2, lng2):
    lat1_rad = np.radians(lat1)
    lng1_rad = np.radians(lng1)
    lat2_rad = np.radians(lat2)
    lng2_rad = np.radians(lng2)
    R = 6371.0 # Earth radius in kilometers
    # Cartesian coordinates
    x1 = R * np.cos(lat1_rad) * np.cos(lng1_rad)
    y1 = R * np.cos(lat1_rad) * np.sin(lng1_rad)
    z1 = R * np.sin(lat1_rad)
    x2 = R * np.cos(lat2_rad) * np.cos(lng2_rad)
    y2 = R * np.cos(lat2_rad) * np.sin(lng2_rad)
    z2 = R * np.sin(lat2_rad)
    # haversine equation
    distances = np.sqrt((x2 - x1)**2 + (y2 - y1)**2 + (z2 - z1)**2)
    return distances
# arrays of lat and lng
